% general parameters
\setchordscheme{
  rotate=270,
  x-unit=13pt, % chord size, x
  y-unit=15pt, % chord size, y
  name-format=\bfseries, % chord name font
  name-below=true,
  finger-format=\bfseries\footnotesize, % fingering font
  tuning={,,,,,},
  restrict-bounding-box=false,
  ringing-style={circle, minimum width=16pt, draw, inner sep=0pt} % this last one doesn't seem to have any effect :(
}

\newsavebox{\A}
\savebox{\A}{
\chordscheme[ % define the chord
  name = A, % chord name
  %position = 1, % first fret position
  %barre = 1/1-6, % fret/string range:label
  finger = {2/4}, % fret/string:label
  finger = {2/3}, % fret/string:label
  finger = {2/2}, % fret/string:label
  ring = {1,5}, % open strings
  mute = {6} % muted strings
]}
